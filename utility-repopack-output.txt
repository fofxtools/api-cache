# Laravel Capsule and Guzzle Caching Helper Functions Library

This library has helper functions for Laravel Capsule, and to cache miscellaneous APIs using Guzzle and MySQL.

## Capsule to Array Conversion

The `get_to_array()` function can be used to convert a Capsule result into an array. For instance:

```php
$result = Capsule::table($table)->where($conditions)->get();
$array = get_to_array($result);
```

## MySQL Guzzle Caching Helper

The src/caching.php file contains the Guzzle_Caching_Helper class, which is the base class that can be extended to cache different APIs. The src/caching_example.php file contains an example of how to extend the base class.

use Illuminate\Database\Capsule\Manager as Capsule;

/**
 * To cache results from the DataForSEO API
 * For the Standard method rather than the Live method, you have to post tasks and specify a pingback URL or a postback URL
 * I prefer the postbacks because pingbacks require an extra step
 * With postbacks, if you post an array of tasks, DFS will ping the postback URL after each task is completed
 * And then you can fetch the results through json_decode(gzdecode(file_get_contents('php://input'), TRUE)
 * But with pingbacks, DFS pings the pingback URL that the task is completed and sends the task ID
 * Then you have to use the task ID to fetch the data from the task get endpoint
 */
class DFS_Caching_Helper extends Guzzle_Caching_Helper
{
    public $api_name = "DFS";
    // 2840 is the United States
    protected $location_code = '2840';
    protected $language_code = 'en';
    protected $device = 'desktop';
    protected $os = 'windows';
    protected $api_url = 'https://api.dataforseo.com';
    protected $pingback_url;
    protected $postback_url;
    public function __construct()
    {
        parent::__construct(...func_get_args());
        $this->input_terms = array('endpoint' => 'string', 'input_term' => 'string');
        $this->output_terms = array('dfs_status_code' => 'integer', 'internal_links_count' => 'integer', 'plain_text_word_count' => 'integer');
        // Use the lowercase of $this->api_name as the ini file section name
        $host = $this->ini[strtolower($this->api_name)]['host'];
        // The pingback URL must contain the $id and $tag. Or else you will get an "empty GET" error.
        // Need to set symlink to use this URL: ln -s /home/[USER]/Utility/public /home/[USER]/public_html/Utility
        // Can modify with: $this->set_property('pingback_url', $url); Or: $this->set_property('postback_url', $url);
        $this->pingback_url = 'https://www.' .  $host . '/Utility/pingback.php?id=$id&tag=$tag';
        $this->postback_url = 'https://www.' . $host . '/Utility/postback.php';
    }
    public function fill_output_values()
    {
        parent::fill_output_values(...func_get_args());
        // $this->output_values['response_body'] may be compressed, so use the original response body to decode the JSON
        $json = json_decode($this->response["transaction['response']->getBody()"], TRUE);
        // Check the tasks->result->items->status_code if instant_pages.
        // Else use $json["status_code"] if set, or set to NULL
        if ($this->input_values['endpoint'] == 'instant_pages') {
            $this->output_values['dfs_status_code'] = $json["tasks"][0]["result"][0]["items"][0]["status_code"];
        } else {
            if (isset($json["status_code"])) {
                $this->output_values['dfs_status_code'] = $json["status_code"];
            } else {
                $this->output_values['dfs_status_code'] = NULL;
            }
        }
        if ($this->output_values['dfs_status_code'] == '200' || $this->output_values['dfs_status_code'] == '206') {
            $this->output_values['internal_links_count'] = $json["tasks"][0]["result"][0]["items"][0]["meta"]["internal_links_count"];
            $this->output_values['plain_text_word_count'] = $json["tasks"][0]["result"][0]["items"][0]["meta"]["content"]["plain_text_word_count"];
        } else {
            $this->output_values['internal_links_count'] = NULL;
            $this->output_values['plain_text_word_count'] = NULL;
        }
    }
    /**
     * Get the onpage for a URL using instant pages
     * @param   string      $onpage_url
     * @param   bool        $enable_javascript
     * @param   bool        $check_spell
     * @return  mixed
     */
    public function get_instant_pages(string $onpage_url, bool $enable_javascript = FALSE, bool $check_spell = FALSE)
    {
        $this->request_url = $this->api_url . "/v3/on_page/instant_pages";
        $this->guzzle_method = "POST";
        // Reset $this->input_values
        $this->input_values = array();
        $this->input_values['endpoint'] = "instant_pages";
        $this->input_values['input_term'] = $onpage_url;
        $post_array = array();
        $post_array[] = array(
            "url" => $onpage_url,
            "enable_javascript" => $enable_javascript,
            "check_spell" => $check_spell

        $headers = array("Authorization" => "Basic " . $this->ini['dfs']['authorization_basic'], "Content-Type" => "application/json");
        $this->guzzle_config = array();
        $this->guzzle_options = array('headers' => $headers, 'json' => $post_array);
        return $this->get_data($this->input_values);
        //$this->response = guzzle_middleware_request('POST', $this->request_url, $config, $options, TRUE);
        //echo "<pre>";
        //echo print_h2($response, TRUE);
        //echo "</pre>";
        //fill_output_values();
        //Capsule::table($this->cache_table)->insert(array_merge($this->input_values, $this->output_values));
    }
    /**
     * Get a live result.
     * To process organic listings:
     *      $result = json_decode($dfs->get_serp_organic_live($input_keyword), TRUE);
     *      $items = $result['tasks'][0]['result'][0]['items'];
     *      foreach ($items as $item) {
     *          if ($item['type'] == 'organic') {
     *              $organic_array[] = $item;
     *          }
     *      }
     * @param   string      $keyword
     * @return  mixed
     */
    public function get_serp_organic_live(string $keyword)
    {
        $this->request_url = $this->api_url . "/v3/serp/google/organic/live/advanced";
        $this->guzzle_method = "POST";
        // Reset $this->input_values
        $this->input_values = array();
        $this->input_values['endpoint'] = "serp_organic_live";
        $this->input_values['input_term'] = $keyword;
        $post_array = array();
        $post_array[] = array(
            "keyword" => mb_convert_encoding($keyword, "UTF-8"),
            "location_code" => $this->location_code,
            "language_code" => $this->language_code,
            "device" => $this->device,
            "os" => $this->os
        );
        $headers = array("Authorization" => "Basic " . $this->ini['dfs']['authorization_basic'], "Content-Type" => "application/json");
        $this->guzzle_config = array();
        $this->guzzle_options = array('headers' => $headers, 'json' => $post_array);
        return $this->get_data($this->input_values);
    }
    /**
     * Post a set of tasks using a postback_url. $keywords is assumed to be an array.
     * Does not interact with the cache array, because that is what the postback does.
     * Instead I have to cache the attempt.
     * See: https://docs.dataforseo.com/v3/serp/google/organic/task_post/
     * Documentation: "You can send up to 2000 API calls per minute, with each POST call containing no more than 100 tasks. If your POST call contains over 100 tasks, the tasks over this limit will return the 40006 error."
     * @param   array       $keywords
     * @return  mixed
     */
    public function serp_google_organic_task_post(array $keywords)
    {
        $this->request_url = $this->api_url . "/v3/serp/google/organic/task_post";
        $this->guzzle_method = "POST";
        $this->input_values['endpoint'] = "serp_google_organic_task_post";
        // The input term is the serialize() of the array of keywords
        $this->input_values['input_term'] = serialize($keywords);
        $post_array = array();
        foreach ($keywords as $keyword) {
            $post_array[] = array(
                "keyword" => mb_convert_encoding($keyword, "UTF-8"),
                "location_code" => $this->location_code,
                "language_code" => $this->language_code,
                "device" => $this->device,
                "os" => $this->os,
                "postback_data" => "advanced",
                "postback_url" => $this->postback_url
            );
        }
        $headers = array("Authorization" => "Basic " . $this->ini['dfs']['authorization_basic'], "Content-Type" => "application/json");
        $this->guzzle_config = array();
        $this->guzzle_options = array('headers' => $headers, 'json' => $post_array);
        return $this->get_data($this->input_values);
    }
    /**
     * Use pingback rather than postback.
     * I prefer postback, since this only gives you the task IDs, and you have to complete another step to get the results.
     * @param   array       $keywords
     * @return  mixed
     */
    public function serp_google_organic_task_post_pingback(array $keywords)
    {
        $this->request_url = $this->api_url . "/v3/serp/google/organic/task_post";
        $this->guzzle_method = "POST";
        $this->input_values['endpoint'] = "serp_google_organic_task_post";
        // Use tag as the input term.
        // Since tag is limited to 255 characters, use sha1(implode()) instead of serialize() for the tag.
        $tag = sha1(implode(', ', $keywords));
        $this->input_values['input_term'] = serialize($keywords);
        $post_array = array();
        foreach ($keywords as $keyword) {
            $post_array[] = array(
                "keyword" => mb_convert_encoding($keyword, "UTF-8"),
                "location_code" => $this->location_code,
                "language_code" => $this->language_code,
                "device" => $this->device,
                "os" => $this->os,
                "tag" => mb_convert_encoding($tag, "UTF-8"),
                "pingback_url" => $this->pingback_url
            );
        }
        $headers = array("Authorization" => "Basic " . $this->ini['dfs']['authorization_basic'], "Content-Type" => "application/json");
        $this->guzzle_config = array();
        $this->guzzle_options = array('headers' => $headers, 'json' => $post_array);
        return $this->get_data($this->input_values);
    }
    /**
     * Cache the postback results.
     * Used in postback.php.
     * @param   string      $post_data_in       The gzencoded data
     */
    public function cache_serp_google_organic_task_post($post_data_in)
    {
        $this->request_url = $this->api_url . "/v3/serp/google/organic/task_post";
        $this->guzzle_method = "POST";
        $this->input_values['endpoint'] = "serp_google_organic_task_post";
        // The post data from DFS is always gz encoded
        $this->output_values['response_body'] = gzdecode($post_data_in);
        $post_data_array = json_decode($this->output_values['response_body'], TRUE);
        $this->input_values['input_term'] = $post_data_array["tasks"][0]["data"]["keyword"];
        $this->output_values['request_url'] = $this->request_url;
        $this->output_values['request_headers'] = NULL;
        $this->output_values['request_body'] = NULL;
        $this->output_values['response_headers'] = NULL;
        // $post_data_in will already by compressed, so compression here is redundant. But I will do it anyway to make consistency easier.
        if ($this->compress) {
            $this->output_values['response_body'] = gzdeflate($post_data_in);
        } else {
            $this->output_values['response_body'] = $post_data_in;
        }
        $this->output_values['status_code'] = NULL;
        $this->output_values['content_length'] = strlen($post_data_in);
        $this->output_values['dfs_status_code'] = $post_data_array['status_code'];
        $this->output_values['internal_links_count'] = NULL;
        $this->output_values['plain_text_word_count'] = NULL;
        $this->output_values['size_data'] = strlen($this->output_values['response_body']);
        // The post data json will say something like: '"time": "0.0818 sec.",'
        // Use strstr to remove everything after a space.
        $this->output_values['response_time'] = strstr($post_data_array["time"], ' ', TRUE);
        $this->cache_data();
    }
    /**
     * After an ID is sent to the pingback_url, fetch the results and cache them
     * @param   int     $id
     * @param   bool    $tag
     * @return  mixed

    public function serp_google_organic_task_ready($id, $tag = FALSE)

        $this->guzzle_config = array();
        $this->guzzle_options = array('headers' => $headers);
        return $this->get_data($this->input_values);
    }

<?php

use Illuminate\Database\Capsule\Manager as Capsule;

/**
 * An example class for Guzzle_Caching_Helper.
 * Takes three variables, and sends them to httpbin.org via GET or POST.
 * An example:
 *     $example = new Example_Caching_Helper();
 *     $example->create_table();
 *     echo $example->get('Hello', 3/7, 'World.');
 *     echo $example->post('Goodbye', 7/3, 'Earth.');
 */
class Example_Caching_Helper extends Guzzle_Caching_Helper
{
    public $api_name = "Example";
    public function __construct()
    {
        parent::__construct(...func_get_args());
        $this->input_terms = array('method' => 'string', 'var1' => 'string', 'var2' => 'double', 'var3' => 'string');
        $this->output_terms = array('vars' => 'text');
    }
    public function fill_output_values()
    {
        parent::fill_output_values(...func_get_args());
        // $this->output_values['response_body'] may be compressed, so use the original response body
        $this->output_values['vars'] = $this->response["transaction['response']->getBody()"];
    }
    public function get(string $var1, float $var2, string $var3)
    {
        $this->guzzle_method = "GET";
        $this->request_url = "https://httpbin.org/response-headers?var1=" . urlencode($var1) . "&var2=" . urlencode($var2) . "&var3=" . urlencode($var3);
        $this->guzzle_config = guzzle_config();
        $this->guzzle_options = guzzle_options();
        $this->null_input_values();
        $this->input_values['method'] = $this->guzzle_method;
        $this->input_values['var1'] = $var1;
        $this->input_values['var2'] = $var2;
        $this->input_values['var3'] = $var3;
        return $this->get_data($this->input_values);
    }
    public function post(string $var1, float $var2, string $var3)
    {
        $this->guzzle_method = "POST";
        $this->request_url = "https://httpbin.org/post";
        $form_params = array('var1' => $var1, 'var2' => $var2, 'var3' => $var3);
        $this->guzzle_config = guzzle_config();
        $this->guzzle_options = guzzle_options();
        $this->guzzle_options['form_params'] = $form_params;
        $this->null_input_values();
        $this->input_values['method'] = $this->guzzle_method;
        $this->input_values['var1'] = $var1;
        $this->input_values['var2'] = $var2;
        $this->input_values['var3'] = $var3;
        return $this->get_data($this->input_values);
    }
}

<?php

use Illuminate\Database\Capsule\Manager as Capsule;
use Spatie\Fork\Fork;
use GuzzleHttp\Client;

/**
 * A caching class for OpenAI
 * This class does not use Guzzle, so the parent get_data() function is overwritten
 * Instead it uses: https://github.com/openai-php/client
 *      composer require openai-php/client
 * See:
 *      https://platform.openai.com/docs/models/model-endpoint-compatibility
 *      https://platform.openai.com/docs/api-reference/completions/create
 */
class OpenAI_Caching_Helper extends Guzzle_Caching_Helper
{
    public $api_name = "OpenAI";
    protected $ini_section = "openai";
    protected $api_key = '';
    protected $client;
    public function __construct()
    {
        parent::__construct(...func_get_args());
        $this->input_terms = array('prompt' => 'text', 'model' => 'string', 'max_tokens' => 'integer', 'temperature' => 'float', 'top_p' => 'float', 'presence_penalty' => 'float', 'frequency_penalty' => 'float');
        $this->output_terms = array('completion' => 'text', 'total_tokens' => 'integer', 'Pegasus Prompt' => 'mediumText', 'Pegasus Completion' => 'mediumText');
        $this->fullText_terms = array('prompt', 'completion', 'Pegasus Prompt', 'Pegasus Completion');
        $this->api_key = $this->ini[$this->ini_section]['api_key'];
        //$this->client = OpenAI::client($this->api_key);
        $timeout = 10;
        $this->client = OpenAI::factory()->withHttpClient(new Client(['connect_timeout' => $timeout, 'timeout'  => $timeout]))->withApiKey($this->api_key)->make();
    }
    /**
     * Overwrite the parent fill_output_values()
     */
    public function fill_output_values()
    {
        // Since the response is an object, calculate the length after serializing it
        $serialized = serialize($this->response);
        $this->output_values['content_length'] = strlen($serialized);
        $this->output_values['response_time'] = $this->elapsed_time;
        if ($this->compress) {
            $this->output_values['response_body'] = gzdeflate($serialized);
        } else {
            $this->output_values['response_body'] = $serialized;
        }
        $this->output_values['size_data'] = strlen($this->output_values['response_body']);
        $this->output_values['completion'] = $this->response['choices'][0]['text'];
        // When I var_dump($result), it seems to be labeled totalTokens, but when I print_r($result['usage']), it is total_tokens
        $this->output_values['total_tokens'] = $this->response['usage']['total_tokens'];
    }
    /**
     * Get the completion for a prompt
     * Replaces exec_guzzle()
     * @param   string      $prompt
     * @param   string      $model
     * @param   int         $max_tokens
     * @param   float       $temperature
     * @param   float       $top_p
     * @param   float       $presence_penalty
     * @param   float       $frequency_penalty
     * @return  string
     */
    public function exec_api(string $prompt, string $model = 'text-ada-001', int $max_tokens = 50, float $temperature = .9, float $top_p = 1, float $presence_penalty = 1, float $frequency_penalty = 1)
    {
        $this->start_time = microtime(TRUE);
        echo date("Y-m-d H:i:s") . " exec_api() - Generating completion for prompt '$prompt', model $model, max_tokens $max_tokens..." . PHP_EOL;
        // The response should be an OpenAI\Responses\Completions\CreateResponse object
        try {
            $this->response = $this->client->completions()->create([
                'model' => $model,
                'prompt' => $prompt,
                'max_tokens' => $max_tokens,
                'temperature' => $temperature,
                'top_p' => $top_p,
                'presence_penalty' => $presence_penalty,
                'frequency_penalty' => $frequency_penalty
            ]);
        } catch (\Exception $e) {
            echo date("Y-m-d H:i:s") . "exec_api() - EXCEPTION, error with prompt '$prompt', message: " . $e->getMessage() . PHP_EOL;
            $this->response = FALSE;
        }
        $this->end_time = microtime(TRUE);
        $this->elapsed_time = $this->end_time - $this->start_time;
        return $this->response;
    }
    /**
     * Since this class doesn't use Guzzle, overwrite the parent get_data() function
     * @param   array       $values_array
     * @return  string
     */
    public function get_data($values_array)
    {
        // If there is only 1 item, make an array of it
        if (!is_array($values_array)) {
            $values_array = array($values_array);
        }
        $result = $this->check_cache($values_array);
        $array = get_to_array($result);
        if (count($array) > 0) {
            if ($this->compress) {
                return gzinflate($array[0]['response_body']);
            } else {
                return $array[0]['response_body'];
            }
        } else {
            $this->response = $this->exec_api($this->input_values['prompt'], $this->input_values['model'], $this->input_values['max_tokens'], $this->input_values['temperature'], $this->input_values['top_p'], $this->input_values['presence_penalty'], $this->input_values['frequency_penalty']);
            if ($this->response) {
                $this->fill_output_values();
                // Only cache if the completion not blank
                if (isset($this->output_values['completion']) && (strlen(trim($this->output_values['completion'])) > 0)) {
                    $this->cache_data();
                    if ($this->compress) {
                        return gzinflate($this->output_values['response_body']);
                    } else {
                        return $this->output_values['response_body'];
                    }
                } else {
                    return '';
                }
            } else {
                return '';
            }
        }
    }
    /**
     * Call get_data() after setting input values based on completion parameters
     * @param   string      $prompt
     * @param   string      $model
     * @param   int         $max_tokens
     * @param   float       $temperature
     * @param   float       $top_p
     * @param   float       $presence_penalty
     * @param   float       $frequency_penalty
     * @return  string
     */
    public function complete(string $prompt, string $model = 'text-ada-001', int $max_tokens = 50, float $temperature = .9, float $top_p = 1, float $presence_penalty = 1, float $frequency_penalty = 1)
    {
        $this->null_input_values();
        $this->input_values['prompt'] = $prompt;
        $this->input_values['model'] = $model;
        $this->input_values['max_tokens'] = $max_tokens;
        $this->input_values['temperature'] = $temperature;
        $this->input_values['top_p'] = $top_p;
        $this->input_values['presence_penalty'] = $presence_penalty;
        $this->input_values['frequency_penalty'] = $frequency_penalty;
        $data = $this->get_data($this->input_values);
        if ($data) {
            $result = unserialize($data);
            return trim($result['choices'][0]['text']);
        } else {
            return FALSE;
        }
    }
    /**
     * Create functions with arrays of prompts and settings
     * @param   array       $prompts
     * @param   array       $model
     * @param   array       $max_tokens
     * @param   array       $temperature
     * @param   array       $presence_penalty
     * @param   array       $frequency_penalty
     */
    public function create_functions(array $prompts, array $model, array $max_tokens, array $temperature, array $top_p, array $presence_penalty, array $frequency_penalty)
    {
        $functions = array();
        for ($i = 0; $i < count($prompts); $i++) {
            $functions[] = function () use ($i, $prompts, $model, $max_tokens, $temperature, $top_p, $presence_penalty, $frequency_penalty) {
                $openai_limit = 2500;
                $completion = $this->complete($prompts[$i], $model[$i], $max_tokens[$i], $temperature[$i], $top_p[$i], $presence_penalty[$i], $frequency_penalty[$i]);
                if ($completion) {
                    $result = array('completion' => $completion, 'prompt' => $prompts[$i], 'model' => $model[$i], 'max_tokens' => $max_tokens[$i], 'temperature' => $temperature[$i], 'top_p' => $top_p[$i], 'presence_penalty' => $presence_penalty[$i], 'frequency_penalty' => $frequency_penalty[$i]);
                    // Sleep to stay within limits
                    limitSleep($openai_limit);
                    return $result;
                } else {
                    return FALSE;
                }
            };
        }
        return $functions;
    }
    /**
     * Generate completions with arrays of parameters
     * @param   array       $prompts
     * @param   mixed       $model
     * @param   mixed       $max_tokens
     * @param   mixed       $temperature
     * @param   mixed       $presence_penalty
     * @param   mixed       $frequency_penalty
     */

            $frequency_penalty = array_fill(0, $count, $frequency_penalty);
        }

 * An example:
 *      $urls = array('http://httpbin.org/headers', 'http://httpbin.org/ip', 'http://httpbin.org/user-agent');
 *      $scraping = new Scraping_Caching_Helper();
 *      $scraping->create_table();

        $this->output_terms = array('contents' => 'text');

     * Create an array of functions to pass to Fork
     * @param   array           $urls
     * @param   bool|string     $proxy
     * @return  array
     */
    public function create_guzzle_functions(array $urls, bool|string $proxy = FALSE)
    {
        $functions = array();
        $this->guzzle_method = "GET";
        $count = count($urls);
        foreach ($urls as $url) {
            $functions[] = function () use ($url, $proxy, $count) {
                // It seems that if too many queries are made simultaneously, I get errors like:
                // "Warning:  Packets out of order. Expected 1 received 205. Packet size=5555252"
                // To prevent this, introduce a random sleep for each function.
                // Sleep time must be different for each function, or they will all be executed at once. Potentially giving MySQL errors.
                $sleep_max = ceil($count * .1);
                randSleep($sleep_max);
                $retval = '';
                $parsed = parse_url($url);
                $base_url = $parsed['scheme'] . '://' . $parsed['host'];
                $request_url = $base_url . $parsed['path'];
                $this->request_url = $request_url;
                $this->null_input_values();
                $this->input_values['input_url'] = $request_url;
                // Use copy of input values to pass to get_data()
                // Just in case intervening code causes functions array to overlap with execution of other functions
                $input_values_copy = $this->input_values;
                $guzzle_config = guzzle_config($proxy);
                $config = [
                    // Base URI is used with relative requests
                    'base_url' => $url,
                    // You can set any number of default request options.
                    'timeout'  => 10,
                ];
                $client_config = array_merge_recursive_distinct($guzzle_config, $config);
                $this->guzzle_config = $client_config;
                $this->guzzle_options = guzzle_options(TRUE);
                return $this->get_data($input_values_copy);
            };
        }
        return $functions;
    }
    /**
     * Scrape the array of URLs
     * @param   array           $urls
     * @param   bool|string     $proxy
     */
    public function scrape_urls(array $urls, bool|string $proxy = FALSE)
    {
        // set: $this->guzzle_method, $this->request_url, $this->guzzle_config, $this->guzzle_options
        // Then in create_guzzle_functions(), exec_guzzle()
        $results = Fork::new()->run(...$this->create_guzzle_functions($urls, $proxy));
        ksort($results);
        return $results;
    }

            if (is_array($input_term)) {

                $count++;

        $function = 'serialize';

function check_semrush_connection($ini_file = FALSE, $section = FALSE)
{
    global $semrush_ini_file, $semrush_ini_file_section;
    if (!$ini_file) {
        $ini_file = $semrush_ini_file;
    }

/**
 * Take associative semrush arrays and combine them where KD is numeric
 * @param   array       ...$arrays
 * @return  array       $merged_filled
 */
function process_semrush_arrays(array ...$arrays)
{
    $merged = array('Keyword' => array());
    $next_int_key = 0;
    $executed_next_int_key = FALSE;
    foreach ($arrays as $array) {
        $fields = array_keys($array);
        for ($i = 0; $i < count($array['Keyword']); $i++) {
            $keyword = $array['Keyword'][$i];
            $kd = $array['Keyword Difficulty'][$i];
            // is_int() returns FALSE for integers stored as type string, so use is_numeric() instead
            if (is_numeric($kd) && !in_array($keyword, $merged['Keyword'])) {
                if (empty($merged['Keyword'])) {
                    $next_int_key = 0;
                } else {
                    if (!$executed_next_int_key) {
                        $next_int_key = next_int_key($merged['Keyword']);
                        $executed_next_int_key = TRUE;
                    } else {
                        $next_int_key = $next_int_key + 1;
                    }
                }
                foreach ($fields as $field) {
                    $merged[$field][$next_int_key] = $array[$field][$i];
                }
            }
        }
    }
    $merged_filled = assoc_array_fill_empty_values($merged, 'Keyword');
    return $merged_filled;
}

/**
 * Take associative semrush arrays and combine them, optionally aggregating the data
 * @param   mixed       ...$arrays
 * @return  array       $array
 */
function process_semrush_arrays_aggregate(mixed ...$arrays)
{
    $combined_fields = array();
    $merged = array();
    // If the first argument is a boolean, assume it saying whether to aggregate all values or not
    if (is_bool($arrays[0])) {
        $aggregate = array_shift($arrays);
    } else {
        $aggregate = FALSE;
    }
    foreach ($arrays as $array) {
        $fields = array_keys($array);
        $combined_fields = array_merge($combined_fields, $fields);
        for ($i = 0; $i < count($array['Keyword']); $i++) {
            $keyword = $array['Keyword'][$i];
            $data = array();
            foreach ($fields as $field) {
                if ($field != 'Keyword') {
                    if ($aggregate) {
                        $data[$field] = array($array[$field][$i]);
                    } else {
                        $data[$field] = $array[$field][$i];
                    }
                }
            }
            if (!isset($merged[$keyword])) {
                $merged[$keyword] = $data;
            } else {
                if ($aggregate) {
                    foreach ($fields as $field) {
                        if ($field != 'Keyword') {
                            if (!isset($merged[$keyword][$field]) || !is_array($merged[$keyword][$field])) {
                                $merged[$keyword][$field] = array();
                            }
                            $merged[$keyword][$field] = array_merge($merged[$keyword][$field], $data[$field]);
                        }
                    }
                }
            }
        }
    }
    $combined_fields = array_unique($combined_fields);
    $array = array();
    $i = 0;
    foreach ($merged as $keyword => $data) {
        $array['Keyword'][$i] = $keyword;
        // Initialize with empty data
        foreach ($combined_fields as $field) {
            if ($field != 'Keyword') {
                $array[$field][$i] = '';
            }
        }
        // Fill with actual data
        foreach ($data as $field => $value) {
            $array[$field][$i] = $value;
        }
        $i++;
    }
    return $array;
}

    for ($i = 0; $i < count($array); $i++) {

            $array = str_csv_to_array($file_contents, ",", TRUE, "\"", "\\", TRUE);
            $assoc = csv_array_to_assoc($array);
            if (!empty($assoc)) {
                $assoc_arrays[] = $assoc;
            }
            //print_r(assoc_array_truncate($assoc, 0, 10));
        }
    }
    $merged = process_semrush_arrays_aggregate(...$assoc_arrays);
    //print_r(assoc_array_truncate($merged, 0, 20));

/**
 * Take associative arrays of aggregate data and generate stats
 * Example:
 *      $data = get_keyword_file_data('accounting');
 *      $stats = semrush_aggregate_stats($data);
 *      print_r(slice($stats, 0, 50));
 * @param   array       $merged
 * @param   boolean     $filter_numeric     Whether to filter by validating that Volume and Keyword Difficulty are numeric
 * @return  array
 */
function semrush_aggregate_stats(array $merged, $filter_numeric = TRUE)
{
    $stats = array();
    if (!isset($merged['Keyword']) || !is_array($merged['Keyword'])) {
        return $stats;
    } else {
        $keys = array_keys($merged['Keyword']);
        for ($i = 0; $i < count($keys); $i++) {
            $index = $keys[$i];
            $keyword = $merged['Keyword'][$index];
            $data = array('Keyword' => $keyword);
            // Choose minimum Volume
            if (is_array($merged['Volume'][$index])) {
                $data['Volume'] = min($merged['Volume'][$index]);
            } else {
                $data['Volume'] = '';
            }
            // Choose maximum Keyword Difficulty
            if (is_array($merged['Keyword Difficulty'][$index])) {
                $data['Keyword Difficulty'] = max($merged['Keyword Difficulty'][$index]);
            } else {
                $data['Keyword Difficulty'] = '';
            }
            // Choose average CPC
            if (is_array($merged['CPC (USD)'][$index]) && array_is_numeric($merged['CPC (USD)'][$index])) {
                $data['CPC (USD)'] = round(array_average($merged['CPC (USD)'][$index]), 2);
            } else {
                $data['CPC (USD)'] = '';
            }
            // Choose average Competitive Density
            if (is_array($merged['Competitive Density'][$index]) && array_is_numeric($merged['Competitive Density'][$index])) {
                $data['Competitive Density'] = round(array_average($merged['Competitive Density'][$index]), 2);
            } else {
                $data['Competitive Density'] = '';
            }
            // Choose average Number of Results
            if (is_array($merged['Number of Results'][$index]) && array_is_numeric($merged['Number of Results'][$index])) {
                $data['Number of Results'] = round(array_average($merged['Number of Results'][$index]), 2);
            } else {
                $data['Number of Results'] = '';
            }
            // Choose the first Intent
            if (is_array($merged['Intent'][$index])) {
                $data['Intent'] = $merged['Intent'][$index][0];
            } else {
                $data['Intent'] = '';
            }
            // Choose the first SERP Features
            if (is_array($merged['SERP Features'][$index])) {
                $data['SERP Features'] = $merged['SERP Features'][$index][0];
            } else {
                $data['SERP Features'] = '';
            }
            // Choose average Trend
            if (is_array($merged['Trend'][$index])) {
                $trend_array = array();
                foreach ($merged['Trend'][$index] as $trend) {
                    $exploded = explode_trim(',', $trend);
                    if ($exploded && count($exploded) == 12) {
                        $trend_array[] = $exploded;
                    }
                }
                // Ensure there are elements to prevent errors
                if (count($trend_array) > 0) {
                    $averages = average_across_arrays($trend_array, 2);
                } else {
                    // implode() will give errors if not an array
                    $averages = array();
                }
                $trend_string = implode(',', $averages);
                $data['Trend'] = $trend_string;
            } else {
                $data['Trend'] = '';
            }
            if (is_numeric($data['Volume']) && is_numeric($data['Keyword Difficulty'])) {
                // Set divisor to minimum 0.5
                // Raise KD to the power of 3 to limit to lower difficulty keywords
                $data['Score'] = $data['Volume'] / pow(max(0.5, $data['Keyword Difficulty']), 3);
            }
            $stats[] = $data;
        }
        // Optionally only include items with numeric Volume and Keyword Difficulty. Note that this may result in an empty stats array.
        if ($filter_numeric) {
            $stats_filtered = array();
            for ($i = 0; $i < count($stats); $i++) {
                if (is_numeric($stats[$i]['Volume']) && is_numeric($stats[$i]['Keyword Difficulty'])) {
                    $stats_filtered[] = $stats[$i];
                }
            }
            return $stats_filtered;
        } else {
            return $stats;
        }
    }
}

/**
 * Get sorted stats for a keyword
 * @depend  get_keyword_file_data()
 * @depend  semrush_aggregate_stats()
 * @depend  array_multisort_by_array()
 * @param   string      $keyword
 * @param   string      $sort_field
 * @param   int         $sort_order
 * @return  array       $stats
 */
function get_stats_sorted($keyword, $sort_field = 'Score', $sort_order = SORT_DESC, $sort_flag = SORT_REGULAR)
{
    $merged = get_keyword_file_data($keyword);
    $stats = semrush_aggregate_stats($merged);
    //print_r(slice($stats));
    $sort_array = array();
    for ($i = 0; $i < count($stats); $i++) {
        $sort_array[] = $stats[$i][$sort_field];
    }
    array_multisort_by_array($stats, $sort_array, $sort_order, $sort_flag);
    return $stats;
}

/**
 * Save stats to a file
 * To delete files: rm -f ../files/*.tsv ../files/*.txt
 * @depend  get_keywords()
 * @depend  get_keyword_file_data()
 * @depend  semrush_aggregate_stats()
 * @depend  array_multisort_by_array()
 * @depend  array_nested_to_tsv()
 * @param   int     $limit
 */
function save_stats_to_file($limit = 6)
{
    $keywords = get_keywords();
    $sort_flag = SORT_REGULAR;
    //$sort_flag = SORT_NATURAL;
    $sort_fields = array('Keyword Difficulty' => SORT_ASC, 'Score' => SORT_DESC, 'Volume' => SORT_DESC);
    // Don't save Intent or SERP Features fields
    $selected_fields = array('Keyword', 'Volume', 'Keyword Difficulty', 'CPC (USD)', 'Competitive Density', 'Number of Results', 'Score');
    $counter = 0;
    $folder = server_home() . '/semrush_php/files/';
    for ($i = 0; ($i < count($keywords)) && ($counter < $limit); $i++) {
        $row_id = $keywords[$i]['row_id'];
        $keyword = $keywords[$i]['keyword'];
        $merged = array();
        $stats = array();
        foreach ($sort_fields as $sort_field => $sort_type) {
            if ($counter < $limit) {
                $keyword_safe = strtolower(strip_non_alnum(trim($keyword)));
                // First check if the sort file was already made
                $sort_field_safe = strtolower(str_replace(' ', '_', $sort_field));
                $filename = $folder . $row_id . "-" . $keyword_safe . "-sort-" . $sort_field_safe . '.tsv';
                // If the file was not made, create it
                if (!file_exists($filename)) {
                    echo date("Y-m-d H:i:s") . " Processing $keyword - $filename..." . PHP_EOL;
                    // Check if merged and stats files exist
                    // Create compressed serialized files if they do not
                    $merged_filename = $folder . "merged-" . $keyword_safe . ".txt";
                    if (file_exists($merged_filename)) {
                        $merged = unserialize(gzinflate(file_get_contents($merged_filename)));
                    } else {
                        $merged = get_keyword_file_data($keyword);
                        file_put_contents($merged_filename, gzdeflate(serialize($merged)));
                    }
                    $stats_filename = $folder . "stats-" . $keyword_safe . ".txt";
                    if (file_exists($stats_filename)) {
                        $stats = unserialize(gzinflate(file_get_contents($stats_filename)));
                    } else {
                        $stats = semrush_aggregate_stats($merged);
                        file_put_contents($stats_filename, gzdeflate(serialize($stats)));
                    }
                    // Save summary stats if this keyword is not already present
                    $summary_filename = $folder . 'Summary.tsv';
                    if (file_exists($summary_filename)) {
                        $summary_contents = file_get_contents($summary_filename);
                    } else {
                        $summary_contents = '';
                    }
                    if (!stristr($summary_contents, PHP_EOL . $keyword . "\t")) {
                        $summary = stats_summary($stats, $keyword);
                        // If first keyword, add headers. Else don't add headers.
                        if ($i === 0) {
                            $contents_summary = implode("\t", array_keys($summary)) . PHP_EOL;
                            $contents_summary .= implode("\t", array_values($summary)) . PHP_EOL;
                        } else {
                            $contents_summary = implode("\t", array_values($summary)) . PHP_EOL;
                        }
                        file_put_contents($summary_filename, $contents_summary, FILE_APPEND);
                    }
                    // If $stats is not empty, create an array of the sort field to sort by
                    // Else create a blank file
                    if (!empty($stats)) {
                        $stats_selected = array();
                        $sort_array = array();
                        for ($j = 0; $j < count($stats); $j++) {
                            $sort_array[] = $stats[$j][$sort_field];
                            foreach ($selected_fields as $selected_field) {
                                $stats_selected[$j][$selected_field] = $stats[$j][$selected_field];
                            }
                        }
                        array_multisort_by_array($stats_selected, $sort_array, $sort_type, $sort_flag);
                        $contents = array_nested_to_tsv($stats_selected, TRUE, TRUE);
                    } else {
                        $contents = '';
                    }
                    file_put_contents($filename, $contents);
                    echo date("Y-m-d H:i:s") . " Saved filename: $filename\tBytes: " . formatBytes(filesize($filename)) . PHP_EOL;

                }
            }
        }

/**
 * Calculate various summary stats for a keyword's semrush_aggregate_stats(get_keyword_file_data($keyword)) result
 * @param   array       $stats
 * @return  array       $summary
 */
function stats_summary(array $stats, $keyword = FALSE, $filter_english = TRUE)
{
    $summary = array();
    if ($keyword) {
        $summary['Keyword'] = $keyword;
    }
    $summary['KD==0'] = 0;
    $summary['Volume>=30'] = 0;
    $summary['KD0, Vol30'] = 0;
    $summary['avg(CPC)'] = '';
    $summary['avg(CD)'] = '';
    $summary['avg(NoR)'] = '';
    $summary['NoR<=100'] = 0;
    $summary['avg(Score)'] = '';
    // If $stats is empty, return the default $summary array
    // $stats may be empty if the keyword has few rows and they were all filtered for not having numeric Volume and KD in semrush_aggregate_stats()
    // Like for "auto audiophilia"
    if (empty($stats)) {
        return $summary;
    } else {
        // If $filter_english, create new array with only English keywords
        if ($filter_english) {
            $stats_edited = array();
            $index = 0;
            for ($i = 0; $i < count($stats); $i++) {
                $is_english = string_is_english_keyboard($stats[$i]['Keyword']);
                if ($is_english) {
                    foreach ($stats[$i] as $field => $value) {
                        $stats_edited[$index][$field] = $value;
                    }
                    $index++;
                }
            }
        } else {
            $stats_edited = $stats;
        }
        // Since we may have filtered out non-English keywords, we need to check again if the array is empty
        if (empty($stats_edited)) {
            return $summary;
        } else {
            for ($i = 0; $i < count($stats_edited); $i++) {
                if ($stats_edited[$i]['Keyword Difficulty'] == 0) {
                    $summary['KD==0']++;
                }
                if ($stats_edited[$i]['Volume'] >= 30) {
                    $summary['Volume>=30']++;
                }
                if ($stats_edited[$i]['Keyword Difficulty'] == 0 && $stats_edited[$i]['Volume'] >= 30) {
                    $summary['KD0, Vol30']++;
                }
                if ($stats_edited[$i]['Number of Results'] <= 100) {
                    $summary['NoR<=100']++;
                }
            }
            // Use tranposed array to calculate averages
            $stats_transposed = transpose_indexed_array($stats_edited);
            $summary['avg(CPC)'] = round(array_average($stats_transposed['CPC (USD)']), 2);
            $summary['avg(CD)'] = round(array_average($stats_transposed['Competitive Density']), 2);
            $summary['avg(NoR)'] = round(array_average($stats_transposed['Number of Results']), 1);
            $summary['avg(Score)'] = round(array_average($stats_transposed['Score']), 2);
            return $summary;
        }
    }
}

/**
 * Comparison function for a keyword's semrush_aggregate_stats(get_keyword_file_data($keyword)) result
 * Or a semrush_aggregate_stats(keyword_array_data($keywords)) result
 * Sort by KD ASC, then Volume DESC, then CPC DESC, then Number of Results ASC
 * Example:
 *      $data = keyword_array_data(array("Apache", "Apache Server", "Apache Software"));
 *      $stats = semrush_aggregate_stats($data);
 *      usort($stats, 'cmp_aggregate_stats');
 *      print_r(slice($stats));
 * @param   array       $aggregate_stats_a
 * @param   array       $aggregate_stats_b
 * @return  int
 */
function cmp_aggregate_stats(array $aggregate_stats_a, array $aggregate_stats_b)
{
    $kd_a = $aggregate_stats_a['Keyword Difficulty'];
    $kd_b = $aggregate_stats_b['Keyword Difficulty'];
    $volume_a = $aggregate_stats_a['Volume'];
    $volume_b = $aggregate_stats_b['Volume'];
    $cpc_a = $aggregate_stats_a['CPC (USD)'];
    $cpc_b = $aggregate_stats_b['CPC (USD)'];
    $nor_a = $aggregate_stats_a['Number of Results'];
    $nor_b = $aggregate_stats_b['Number of Results'];
    if ($kd_a == $kd_b) {
        if ($volume_a == $volume_b) {
            if ($cpc_a == $cpc_b) {
                if ($nor_a == $nor_b) {
                    return 0;
                } else {
                    // Sort by Number of Results ASC
                    return ($nor_a < $nor_b) ? -1 : 1;
                }
            } else {
                // Sort by CPC DESC
                return ($cpc_a > $cpc_b) ? -1 : 1;
            }
        } else {
            // Sort by Volume DESC
            return ($volume_a > $volume_b) ? -1 : 1;
        }
    } else {
        // Sort by KD ASC
        return ($kd_a < $kd_b) ? -1 : 1;
    }
}

/**
 * Comparison function for a keyword's semrush_aggregate_stats(get_keyword_file_data($keyword)) result
 * Or a semrush_aggregate_stats(keyword_array_data($keywords)) result
 * Sort by Score DESC
 * Example:
 *      $data = keyword_array_data(array("Apache", "Apache Server", "Apache Software"));
 *      $stats = semrush_aggregate_stats($data);

 *      print_r(slice($stats));
 * @param   array       $aggregate_stats_a
 * @param   array       $aggregate_stats_b
 * @return  int
 */
function cmp_aggregate_stats_score(array $aggregate_stats_a, array $aggregate_stats_b)
{
    $score_a = $aggregate_stats_a['Score'];
    $score_b = $aggregate_stats_b['Score'];
    if ($score_a == $score_b) {
        return 0;
    } else {
        return ($score_a > $score_b) ? -1 : 1;
    }
}

/**
 * Connect to Capsule
 * @param   string      $ini_file
 */
function capsule_connect($ini_file = 'config.ini.php', $section = 'mysql')
{
    // getcwd() will tell parse_ini_file to use the directory of the file that is including vendor/autoload.php
    // Thus, there must be a config.ini.php in the same folder as the main file
    // Or the relative path must be passed through the arguments
    if ($ini_file == 'config.ini.php') {
        $ini_file = getcwd() . "/" . $ini_file;
    }
    $ini = parse_ini_file($ini_file, TRUE);
    $capsule = new Capsule;
    $capsule->addConnection([
        'driver' => 'mysql',
        'host' => $ini[$section]['host'],
        'database' => $ini[$section]['database'],
        'username' => $ini[$section]['username'],
        'password' => $ini[$section]['password']
    ]);
    $capsule->setAsGlobal();
    $capsule->bootEloquent();
}

        if ($database === $database_name) {
            return TRUE;
        } else {
            return FALSE;
        }
    } else {
        return TRUE;
    }
}
